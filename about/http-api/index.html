HTTP API

更新时间: 2017-10-30

Basement File 的 API 完全居于 HTTP 提供，你可以使用任何你喜欢的语言使用我们的服务。对于 Java和 Node 的用户，我们强烈建议使用已封装好的 SDK.

示例

请求

const urllib = require(‘co-urllib’);
const co = require(‘co’);



co(function*(){
    // 线上的 AK 的话, 对应地址为: https://basement-gzone.alipay.com
    const ret = yield urllib.request(‘http://basement-api.stable.alipay.net/api/1.0/files/public?fileName=1.txt&targetEnv=DEV',{
        method: ‘POST’,
        headers: {
            ‘x-base-id’: ‘5885827c9a7934057979d4db’,
            ‘x-base-masterKey’: ‘1csgxwXVuI_t05UK_EheY7NN’
        },
        data: new Buffer(‘stream’),
        dataType: ‘json’
    });
    console.log(ret.data);
}).catch(e => {
    console.log(e);
})

响应

{
    "ETag":"F7B44CFAFD5C52223D5498196C8A2E7B",
    "name":"1.txt",
    "size":6,
    "type":"txt",
    "url":"http://alipay-rmsdeploy-dev-image.oss-cn-hangzhou-zmf.aliyuncs.com/lanjuetest/hDPwImZLYwLIOmzEBRjR.txt"
}

const co = require(‘co’);

    const ret = yield urllib.request(‘http://basement-api.stable.alipay.net/api/1.0/files/remove',{
        method: ‘DELETE’,
        headers: {
            ‘x-base-id’: ‘5885827c9a7934057979d4db’,

        dataType: ‘json’
    });
    console.log(JSON.stringify(ret.data));
}).catch(e => {
    console.log(e);
})

响应

{
    "success":true
}
